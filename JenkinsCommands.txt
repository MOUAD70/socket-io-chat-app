# Récupération de l’image Jenkins
docker pull jenkins/jenkins:lts

# Lancement de Jenkins avec docker
docker run -d --name jenkins -p 8080:8080 -p 50000:50000 -v jenkins_home:/var/jenkins_home jenkins/jenkins:lts

# Accès à Jenkins et récupération du mot de passe initial
docker exec jenkins cat /var/jenkins_home/secrets/initialAdminPassword

# Installer Docker dans Jenkins
docker exec -it jenkins bash
apt update
apt install docker.io -y

# Ajouter Jenkins au groupe docker
usermod -aG docker jenkins

# Modifier les permission au niveau du conteneur Jenkins
docker exec -u root jenkins-server chmod 666 /var/run/docker.sock
